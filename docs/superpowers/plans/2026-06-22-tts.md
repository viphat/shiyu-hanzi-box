# TTS (Text-to-Speech) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a one-click pronounce button to Word cards using the browser-native `SpeechSynthesis` API.

**Architecture:** A pure `lib/tts.ts` module wraps `window.speechSynthesis` for voice selection, playback, and state tracking. A `SpeakButton.tsx` React component provides idle/speaking UI with polling. Three existing components embed the button. One i18n key is added.

**Tech Stack:** Web Speech API (`window.speechSynthesis`), React `useState`/`useEffect`, lucide-react `Volume2` icon, Vitest.

### Task 2: Implement `lib/tts.ts`

**Files:**
- Create: `lib/tts.ts`

- [ ] **Step 1: Create the TTS module**

Create `lib/tts.ts`:

```ts
export type TtsState = 'unavailable' | 'idle' | 'speaking';

let state: TtsState = 'idle';
let chineseVoice: SpeechSynthesisVoice | null = null;

function pickChineseVoice(voices: SpeechSynthesisVoice[]): SpeechSynthesisVoice | null {
  const zhCN = voices.find((v) => v.lang === 'zh-CN');
  if (zhCN) return zhCN;
  return voices.find((v) => v.lang.startsWith('zh')) ?? null;
}

/**
 * Initialize the voice list and cache a Chinese voice.
 * Must be called after `getVoices()` resolves (or on `voiceschanged`).
 */
export function initVoices(): void {
  const voices = speechSynthesis.getVoices();
  chineseVoice = pickChineseVoice(voices);

  if (chineseVoice) {
    // Warm up: prime the engine to avoid first-utterance silence bug.
    const warmup = new SpeechSynthesisUtterance(' ');
    warmup.voice = chineseVoice;
    warmup.volume = 0;
    warmup.rate = 10;
    warmup.onend = () => speechSynthesis.cancel();
    speechSynthesis.speak(warmup);
  }
}

export function isChineseVoiceAvailable(): boolean {
  return chineseVoice !== null;
}

export function getTtsState(): TtsState {
  return state;
}

export function speak(text: string): void {
  if (!chineseVoice) return;

  const utterance = new SpeechSynthesisUtterance(text);
  utterance.voice = chineseVoice;

  utterance.onend = () => {
    state = 'idle';
  };
  utterance.onerror = () => {
    state = 'idle';
  };

  state = 'speaking';
  speechSynthesis.speak(utterance);
}

export function stop(): void {
  speechSynthesis.cancel();
  state = 'idle';
}
```

### Task 3: Create `SpeakButton.tsx`

- [ ] **Step 1: Create the SpeakButton component**

Create `entrypoints/newtab/components/SpeakButton.tsx`:

```tsx
import { useEffect, useState } from 'react';
import { Volume2 } from 'lucide-react';
import { t } from '@/lib/i18n';
import {
  getTtsState,
  initVoices,
  isChineseVoiceAvailable,
  speak,
  stop,
} from '@/lib/tts';
import type { TtsState } from '@/lib/tts';
import type { UiLocale } from '@/lib/types';

export function SpeakButton({ text, locale }: { text: string; locale: UiLocale }) {
  const [available, setAvailable] = useState(isChineseVoiceAvailable);
  const [ttsState, setTtsState] = useState<TtsState>(getTtsState);

  useEffect(() => {
    // Initialize voices if not yet done
    const voices = speechSynthesis.getVoices();
    if (voices.length > 0) {
      initVoices();
      setAvailable(isChineseVoiceAvailable());
    }

    const handleVoicesChanged = () => {
      initVoices();
      setAvailable(isChineseVoiceAvailable());
    };
    speechSynthesis.addEventListener('voiceschanged', handleVoicesChanged);

    // Poll ttsState every 200ms to react to speech engine events
    const interval = setInterval(() => {
      setTtsState(getTtsState());
    }, 200);

    return () => {
      speechSynthesis.removeEventListener('voiceschanged', handleVoicesChanged);
      clearInterval(interval);
    };
  }, []);

  if (!available) return null;

  const isSpeaking = ttsState === 'speaking';

  const handleClick = (event: React.MouseEvent) => {
    event.stopPropagation();
    if (isSpeaking) {
      stop();
    } else {
      speak(text);
    }
  };

  return (
    <button
      onClick={handleClick}
      title={t(locale, 'tts.speak')}
      className={`rounded-sm p-1 transition ${isSpeaking ? 'animate-pulse text-cinnabar' : 'text-muted hover:bg-paper-input hover:text-cinnabar'}`}
    >
      <Volume2 className="h-3.5 w-3.5" />
    </button>
  );
}
```
